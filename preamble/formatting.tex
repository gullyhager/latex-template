% filepath: preamble/formatting.tex

% ── Equation Numbering ────────────────────────────────────────
\counterwithin*{equation}{section}

% ── Spacing ───────────────────────────────────────────────────
\setstretch{1.2}

% ── Widow & Orphan Control ────────────────────────────────────
\widowpenalty=10000
\clubpenalty=10000

% ── Hyperlink Styling ─────────────────────────────────────────
\hypersetup{
    colorlinks  = true,
    linkcolor   = black,
    citecolor   = black,
    urlcolor    = black!70
}

% ── Title Block Styling ───────────────────────────────────────
\usepackage{titling}

\setlength{\droptitle}{-3em}          % pull title up toward top of page

\pretitle{\begin{center}\large\bfseries\scshape}
\posttitle{\end{center}\vspace{-0.5em}}

\preauthor{\begin{center}\normalsize}
\postauthor{\end{center}\vspace{-0.8em}}

\predate{\begin{center}\small\itshape}
\postdate{\end{center}\vspace{-0.5em}}

% ── Section Title Styling ─────────────────────────────────────
\usepackage{titlesec}

\titleformat{\section}
    {\normalsize\bfseries\scshape}
    {\thesection}{1em}{}
    [\vspace{0.5pt}\titlerule\vspace{2pt}]

\titleformat{\subsection}
    {\normalsize\bfseries}
    {\thesubsection}{1em}{}

\titleformat{\subsubsection}
    {\normalsize\itshape}
    {\thesubsubsection}{1em}{}

% ── Header & Footer ───────────────────────────────────────────
\usepackage{fancyhdr}

\pagestyle{fancy}
\fancyhf{}
\fancyhead[L]{\small\scshape\leftmark}
\fancyhead[R]{\small\thepage}
\renewcommand{\headrulewidth}{0.4pt}

% Suppress header on the first page
\fancypagestyle{firstpage}{
    \fancyhf{}
    \renewcommand{\headrulewidth}{0pt}
    \fancyfoot[C]{\small\thepage}
}
